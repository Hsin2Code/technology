A -> A : Select
activate A
A -> K : System call - kernel context switch
activate K

deactivate K
note over A : Application blocked
     K <-] : Read response
== Datagram ready ==
activate K
A <- K : Select - data available (readable)
deactivate K
deactivate A

A -> A : Read
activate A
A -> K : System call - kernel context switch
activate K
A <- K : Data movement from kernel space to user space
deactivate K
A -> A : Return
deactivate A
